SELECT room.id AS room_id, room.name AS room_name, room.description AS room_description, room.created_at AS room_created_at, user.id AS user_id, user.first_name AS user_first_name, user.last_name AS user_last_name, user.email AS user_email, user.password AS user_password FROM room INNER JOIN user ON room.user_id = user.id WHERE (name = :where_0) LIMIT 5
